### Elbow

```js
import { Component as CarbonComponent } from '@carbon/charts'
const linkSource = { x: 0, y: 0 }
const linkTarget = { x: 400, y: 0 }
const path = CarbonComponent.buildElbowPathString(linkSource, linkTargetCurve)
// path = 'M0,0L200,0L200,200L400,200'
```

<Canvas style={{ background: '#fff' }}>
  <svg height="260" width="800">
		<g transform="translate(16,16)">
			<Edge path={'M0,0L200,0L200,200L400,200'} />
		</g>
	</svg>
</Canvas>

### Bezier

```js
import { Component as CarbonComponent } from '@carbon/charts'
const linkSource = { x: 0, y: 0 }
const linkTarget = { x: 400, y: 0 }
const path = CarbonComponent.buildBezierPathString(linkSource, linkTargetCurve, 150, 280, 150, 30)
// path = 'M0,0C150,280,150,30,400,200'
```
